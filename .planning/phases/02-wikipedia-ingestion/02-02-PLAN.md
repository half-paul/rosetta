```typescript
import { createHash } from 'node:crypto'

export function computeContentHash(plainText: string): string {
  return createHash('sha256')
    .update(plainText, 'utf8')
    .digest('hex')
    .slice(0, 12)
}

export function buildStableId(
  sectionPath: string,
  contentHash: string,
  revisionId: number,
): string {
  return `${sectionPath}:${contentHash}:${revisionId}`
}

export function normalizeSectionPath(headings: string[]): string {
  return headings
    .map(h => h.toLowerCase().replace(/\s+/g, '_'))
    .join('/')
}
```

       - Define strip selectors constant (per D-08):
         ```typescript
         const STRIP_SELECTORS = [
           '.infobox', '.navbox', '.reflist', '#toc',
           '.mw-editsection', '.ambox', '.tmbox',
           '.reference', '.mw-references-wrap', 'style', 'script',
         ]
         ```

       - Export `function parseWikipediaHtml(html: string, revisionId: number): ParsedSection[]`:
         1. Create JSDOM instance: `const { document } = new JSDOM(html).window`
         2. Strip non-content elements: iterate STRIP_SELECTORS, call `document.querySelectorAll(sel).forEach(el => el.remove())`
         3. Find root: `const root = document.querySelector('.mw-parser-output')` — if null, return empty array
         4. Initialize lead section context (Pitfall 3 — lead before first heading):
            ```typescript
            let currentHeadings: string[] = ['lead']
            let currentSection: ParsedSection = {
              title: 'lead', path: 'lead', position: 0, paragraphs: [],
            }
            const sections: ParsedSection[] = [currentSection]
            let sectionPos = 0
            let paraPos = 0
            ```
         5. Walk children of root (`Array.from(root.children)`):
            - If child has class `mw-heading` (MediaWiki heading wrapper div):
              - Find `h2, h3, h4, h5, h6` inside it
              - If no heading element found, `continue`
              - Extract level: `parseInt(heading.tagName[1], 10)` (2-6)
              - Extract title: `heading.textContent?.trim() ?? ''`
              - Build heading stack based on level:
                - `level === 2`: reset stack to `[title]`
                - `level === 3`: `[currentHeadings[0] ?? title, title]`
                - `level > 3`: `[...currentHeadings.slice(0, level - 2), title]`
              - Create new section: `path = normalizeSectionPath(currentHeadings)`, increment sectionPos, reset paraPos to 0
              - Push new section to sections array, set as currentSection
            - Else if child tagName is `'P'`:
              - Extract `plainText = child.textContent?.trim() ?? ''`
              - Skip if `plainText.length < 10` (Pitfall 6: spacers/empties)
              - Compute: `contentHash = computeContentHash(plainText)`
              - Compute: `stableId = buildStableId(currentSection.path, contentHash, revisionId)`
              - Push to currentSection.paragraphs with position `paraPos++`
         6. Return `sections.filter(s => s.paragraphs.length > 0)` (exclude empty sections)

|-----------|----------|-----------|-------------|-----------------|
| T-02-04 | Tampering | parseWikipediaHtml | mitigate | JSDOM runs in Node.js without script execution by default (scripts are not evaluated). The parser only reads textContent and DOM structure — no innerHTML is ever inserted into the application. All output is plain text strings. |
| T-02-05 | DoS | parseWikipediaHtml | mitigate | Articles with extreme paragraph counts (>500) could create excessive DB rows. The ingestion worker (Plan 03) should cap paragraph count.
