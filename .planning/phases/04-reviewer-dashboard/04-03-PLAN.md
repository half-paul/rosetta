    **2. Replace src/app/dashboard/page.tsx** — Server Component. REMOVE the existing topnav (now in layout). Add queue list:
    ```typescript
    import { getQueue, getQueueDepth, getReviewers } from '@/features/reviews'
    import { QueueFilters } from '@/components/dashboard/queue-filters'
    import { QueueItemRow } from '@/components/dashboard/queue-item-row'
    import { QueueDepthAlert } from '@/components/dashboard/queue-depth-alert'

    export default async function DashboardPage({
      searchParams,
    }: {
      searchParams: Promise<{ status?: string; severity?: string; assignee?: string; page?: string }>
    }) {
      const params = await searchParams // MUST await — Next.js 16

      const items = await getQueue({
        status: params.status,
        severity: params.severity,
        assignee: params.assignee,
        page: params.page ? parseInt(params.page, 10) : 0,
      })
      const depth = await getQueueDepth()
      const reviewers = await getReviewers()

      return (
        <div className="p-6">
          <QueueDepthAlert depth={depth} />
          <h1 className="text-xl font-semibold mb-4">Review queue</h1>
          <QueueFilters reviewers={reviewers} />
          <div className="mt-4 border rounded-lg divide-y">
            {items.length === 0 ? (
              <div className="p-8 text-center">
                <h2 className="text-lg font-semibold">Queue is clear</h2>
                <p className="text-sm text-muted-foreground mt-1">
                  No items waiting for review. New AI-flagged content will appear here as articles are ingested.
                </p>
              </div>
            ) : (
              items.map(item => (
                <QueueItemRow key={item.commentaryId} item={item} reviewers={reviewers} />
              ))
            )}
          </div>
          {/* Pagination controls */}
          {items.length === 25 && (
            <div className="mt-4 flex justify-center">
              <a href={`/dashboard?${new URLSearchParams({ ...params, page: String((parseInt(params.page || '0', 10)) + 1) })}`}
                 className="text-sm text-muted-foreground hover:text-foreground">
                Next page
              </a>
            </div>
          )}
        </div>
      )
    }
    ```
