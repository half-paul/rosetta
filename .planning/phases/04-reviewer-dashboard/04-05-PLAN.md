    **3. Create src/app/dashboard/activity/page.tsx** — Server Component:
    ```typescript
    import { auth } from '@/auth'
    import { redirect } from 'next/navigation'
    import { getGlobalAuditFeed, getReviewers } from '@/features/reviews'
    import { ActivityFeed } from '@/components/dashboard/activity-feed'

    export default async function ActivityPage({
      searchParams,
    }: {
      searchParams: Promise<{ reviewer?: string; action?: string; page?: string }>
    }) {
      const session = await auth()
      if (!session?.user) redirect('/login')

      const params = await searchParams
      const entries = await getGlobalAuditFeed({
        reviewer: params.reviewer,
        action: params.action,
        page: params.page ? parseInt(params.page, 10) : 0,
      })

      return (
        <div className="p-6">
          <h1 className="text-xl font-semibold mb-4">Activity log</h1>
          {/* Filter controls — simple link-based for now */}
          <div className="flex gap-3 mb-4 text-sm">
            <a href="/dashboard/activity" className="text-muted-foreground hover:text-foreground">All</a>
            <a href="/dashboard/activity?action=approve" className="text-muted-foreground hover:text-foreground">Approvals</a>
            <a href="/dashboard/activity?action=reject" className="text-muted-foreground hover:text-foreground">Rejections</a>
            <a href="/dashboard/activity?action=flag" className="text-muted-foreground hover:text-foreground">Flags</a>
          </div>
          <ActivityFeed entries={entries} />
        </div>
      )
    }
    ```
